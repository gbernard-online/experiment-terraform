```bash
$ cat >main.tf <<EOF
variable "color" {
  type    = string
  default = "green"
}

resource "null_resource" "color" {
  triggers = {
    color = var.color
  }
  provisioner "local-exec" {
    command = "echo '\${var.color}' | tee color.txt"
  }
}

output "color" {
  value = var.color
}
EOF

$ terraform fmt --check --diff
```
